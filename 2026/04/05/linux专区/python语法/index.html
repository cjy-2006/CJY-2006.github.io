方法 作用 语法 返回值类型
append() 末尾添加一个元素 list.append(x) None
extend() 末尾追加可迭代对象 list.extend(iterable) None
insert() 指定索引插入元素 list.insert(index, x) None
remove() 删除第一个匹配元素 list.remove(x) None
pop() 删除并返回指定索引元素 list.pop(index=-1) 任意类型
clear() 清空列表 list.clear() None
index() 返回元素第一个匹配索引 list.index(x, start, end) int
count() 统计元素出现次数 list.count(x) int
sort() 原地排序 list.sort(key=None, reverse=False) None
reverse() 原地反转 list.reverse() None
copy() 浅拷贝列表 list.copy() list

方法 作用 语法 返回值类型
find() 查找子串索引，不存在返回-1 str.find(sub, start, end) int
index() 查找子串索引，不存在报错 str.index(sub, start, end) int
count() 统计子串出现次数 str.count(sub, start, end) int
replace() 替换子串，返回新字符串 str.replace(old, new, count) str
split() 按分隔符分割 str.split(sep=None, maxsplit=-1) list
join() 连接可迭代对象 str.join(iterable) str
strip() 去除首尾空白/指定字符 str.strip(chars=None) str
lstrip() 去除左侧空白 str.lstrip() str
rstrip() 去除右侧空白 str.rstrip() str
upper() 转为大写 str.upper() str
lower() 转为小写 str.lower() str
capitalize() 首字母大写 str.capitalize() str
title() 每个单词首字母大写 str.title() str
startswith() 判断是否以指定前缀开头 str.startswith(prefix) bool
endswith() 判断是否以指定后缀结尾 str.endswith(suffix) bool
isdigit() 判断是否全为数字 str.isdigit() bool
isalpha() 判断是否全为字母 str.isalpha() bool
isalnum() 判断是否为字母+数字 str.isalnum() bool
isspace() 判断是否全为空白 str.isspace() bool
center() 居中填充 str.center(width, fillchar) str
ljust() 左对齐填充 str.ljust(width) str
rjust() 右对齐填充 str.rjust(width) str
